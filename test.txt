FROM ubuntu

RUN apt-get update && apt-get -y install mysql-client mysql-server
RUN sed -i ‘s/^bind-address/#bind-adress/g’ /etc/mysql/my.cnf
RUN /etc/init.d/mysql start && echo “GRANT ALL ON *.* TO bengak@’%’ IDENTIFIED BY ‘bengak’ WITH GRANT OPTION; FLUSH PRIVILEGES” | mysql -u root

EXPOSE 3307

CMD [“/usr/bin/mysqld_safe”]
